$('#gallery').cycle({
    fx:     'scrollHorz', 
    speed:   500, 
    timeout: 8000,
    easing: 'easeInOutQuad',
    next: '#gallery-next',
    prev: '#gallery-prev',
    pager: '#gallery-thumbs', 
    pagerAnchorBuilder: function(idx, slide) { 
        return '#gallery-thumbs a:eq(' + idx + ')'; 
    },
    before: function(currSlideElement, nextSlideElement, options, forwardFlag) {
        $('#gallery').animate({'height':$(nextSlideElement).height()}, 500);

  • Placeholderfix

    • Developed personally to improvise HTML5 native placeholder functionality
      1. Based off of jQuery-placeholder plugin
    • Modifcations:
      1. targets a single <form></form> element, finds all text inputs and textareas, toggles user set placeholder text

  • How to Build jQuery Plugins

  • Basic Structure

    Always wrap your plugin as follows:

                
(function( $ ){

	$.fn.myPlugin = function() {
      
		//do plugin stuff
    
	};
    
})( jQuery );

    Allows you to use the $ sign inside your plugin without worrying about conflicts with other JS libraries.

                
$('#element').myPlugin();

  • Context

    Don't redundantly wrap the this keyword in the immediate scope of your plugin's function

                
(function( $ ){

    $.fn.myPlugin = function() {

	//don't call $(this) right here!
        
        this.fadeIn('normal', function(){
        
            //do plugin stuff
        
        });
    
    };
    
})( jQuery );

  • Basic Plugin

    
(function( $ ){
    
    $.fn.maxHeight = function() {
    
    	var max = 0;
    
        this.each(function() {
            max = Math.max( max, $(this).height() );
        });
    
    	return max;
    };
    
})( jQuery );

var tallest = $('div').maxHeight(); 
// Returns the height of the tallest div

  • Chainability

    Unless returning a value, always return the this keyword to maintain chainability.

                
(function( $ ){
    
    $.fn.myPlugin = function( type ) {  
    
        return this.each(function() {
        
        	//do plugin stuff
        
        });
    
    };
    
})( jQuery );

$('div').myPlugin().css('color', 'red');

  • Defaults and Options

                
(function( $ ){

    $.fn.tooltip = function( options ) {  
    
        var settings = {
        	'location'         : 'top',
        	'background-color' : 'blue'
        };
        
        return this.each(function() {        
            if ( options ) { 
            	$.extend( settings, options );
            }
            // Tooltip plugin code here
        });
    
    };
    
})( jQuery );

$('div').tooltip({
	'location' : 'left'
});
